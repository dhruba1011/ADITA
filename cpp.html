4. Largest of Three Numbers

Find the largest among three numbers.

#include <iostream>
using namespace std;

int main() {
    int a, b, c;
    cout << "Enter 3 numbers: ";
    cin >> a >> b >> c;
    if (a >= b && a >= c)
        cout << "Largest: " << a;
    else if (b >= c)
        cout << "Largest: " << b;
    else
        cout << "Largest: " << c;
    return 0;
}

5. Factorial of a Number

Calculate the factorial using a loop.

#include <iostream>
using namespace std;

int main() {
    int n;
    long long fact = 1;
    cout << "Enter a number: ";
    cin >> n;
    for (int i = 1; i <= n; i++) {
        fact *= i;
    }
    cout << "Factorial: " << fact;
    return 0;
}

7. Reverse a Number

Reverse an integer number.

#include <iostream>
using namespace std;

int main() {
    int n, rev = 0;
    cout << "Enter a number: ";
    cin >> n;
    while (n != 0) {
        rev = rev * 10 + (n % 10);
        n /= 10;
    }
    cout << "Reversed Number: " << rev;
    return 0;
}

8. Check Prime Number

Check whether a number is prime.

#include <iostream>
using namespace std;

int main() {
    int n, i;
    bool isPrime = true;
    cout << "Enter a number: ";
    cin >> n;
    if (n <= 1) isPrime = false;
    for (i = 2; i <= n / 2; i++) {
        if (n % i == 0) {
            isPrime = false;
            break;
        }
    }
    if (isPrime)
        cout << "Prime";
    else
        cout << "Not Prime";
    return 0;
}

9. Array Sum

Input 5 integers into an array and print their sum.

#include <iostream>
using namespace std;

int main() {
    int arr[5], sum = 0;
    cout << "Enter 5 numbers:\n";
    for (int i = 0; i < 5; i++) {
        cin >> arr[i];
        sum += arr[i];
    }
    cout << "Sum: " << sum;
    return 0;
}

11. Simple Class and Object

Create a class to input and display student name and age.

#include <iostream>
using namespace std;

class Student {
public:
    string name;
    int age;

    void input() {
        cout << "Enter name and age: ";
        cin >> name >> age;
    }

    void display() {
        cout << "Name: " << name << ", Age: " << age;
    }
};

int main() {
    Student s;
    s.input();
    s.display();
    return 0;
}

12. File Handling: Write and Read File

Write a string to a file and read it back.

#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ofstream outFile("example.txt");
    outFile << "C++ file handling example";
    outFile.close();

    ifstream inFile("example.txt");
    string text;
    getline(inFile, text);
    cout << "File content: " << text;
    inFile.close();
    return 0;
}
